116 InputResult InputBox::messageFmt(char const * titleText, char const * buttonCancelText, char const * buttonOKText, const char * format, ...)
117 {
118  auto r = InputResult::Cancel;
119  va_list ap;
120  va_start(ap, format);
121  int size = vsnprintf(nullptr, 0, format, ap) + 1;
122  if (size > 0) {
123  va_end(ap);
124  va_start(ap, format);
125  char buf[size + 1];
126  vsnprintf(buf, size, format, ap);
127  r = message(titleText, buf, buttonCancelText, buttonOKText);
128  }
129  va_end(ap);
130  return r;
131 }

235 void InputApp::init()
236 {
237  retval = InputResult::None;
238 
239  rootWindow()->frameStyle().backgroundColor = backgroundColor;
240  font = &FONT_std_14;
241 
242  const int titleHeight = titleText && strlen(titleText) ? font->height : 0;
243  const bool buttonsExist = buttonCancelText || buttonOKText;
244  const int buttonCancelExtent = buttonCancelText ? canvas()->textExtent(font, buttonCancelText) + 10 : 0;
245  const int buttonOKExtent = buttonOKText ? canvas()->textExtent(font, buttonOKText) + 10 : 0;
246  const int buttonsWidth = imax(imax(buttonCancelExtent, buttonOKExtent), 40);
247  const int totButtons = (buttonCancelExtent ? 1 : 0) + (buttonOKExtent ? 1 : 0);
248  const int buttonsHeight = buttonsExist ? font->height + 6 : 0;
249  constexpr int buttonsSpace = 10;
250 
251  requiredWidth = buttonsWidth * totButtons + (2 * buttonsSpace) * totButtons;
252  requiredHeight = buttonsHeight + titleHeight + font->height * 2 + 5;
253 
254  calcRequiredSize();
255 
256  requiredWidth = imin(requiredWidth, canvas()->getWidth());
257 
258  mainFrame = new uiFrame(rootWindow(), titleText, UIWINDOW_PARENTCENTER, Size(requiredWidth, requiredHeight), false);
259  mainFrame->frameProps().resizeable = false;
260  mainFrame->frameProps().hasMaximizeButton = false;
261  mainFrame->frameProps().hasMinimizeButton = false;
262  mainFrame->frameProps().hasCloseButton = false;
263  mainFrame->onKeyUp = [&](uiKeyEventInfo key) {
264  if (key.VK == VK_RETURN || key.VK == VK_KP_ENTER) {
265  retval = InputResult::Enter;
266  finalize();
267  } else if (key.VK == VK_ESCAPE) {
268  retval = InputResult::Cancel;
269  finalize();
270  }
271  };
272 
273  autoOKLabel = nullptr;
274 
275  uiWindow * controlToFocus = nullptr;
276 
277  if (buttonsExist) {
278 
279  // setup panel (where buttons are positioned)
280 
281  int panelHeight = buttonsHeight + 10;
282  panel = new uiPanel(mainFrame, Point(mainFrame->clientPos().X - 1, mainFrame->clientPos().Y + mainFrame->clientSize().height - panelHeight), Size(mainFrame->clientSize().width + 2, panelHeight));
283  panel->windowStyle().borderColor = RGB888(128, 128, 128);
284  panel->panelStyle().backgroundColor = mainFrame->frameStyle().backgroundColor;
285 
286  // setup buttons
287 
288  int y = (panelHeight - buttonsHeight) / 2;
289  int x = panel->clientSize().width - buttonsWidth * totButtons - buttonsSpace * totButtons; // right aligned
290 
291  if (buttonCancelText) {
292  auto buttonCancel = new uiButton(panel, buttonCancelText, Point(x, y), Size(buttonsWidth, buttonsHeight));
293  buttonCancel->onClick = [&]() {
294  retval = InputResult::Cancel;
295  finalize();
296  };
297  x += buttonsWidth + buttonsSpace;
298  controlToFocus = buttonCancel;
299  }
300 
301  if (buttonOKText) {
302  auto buttonOK = new uiButton(panel, buttonOKText, Point(x, y), Size(buttonsWidth, buttonsHeight));
303  buttonOK->onClick = [&]() {
304  retval = InputResult::Enter;
305  finalize();
306  };
307  controlToFocus = buttonOK;
308  }
309 
310  if (autoOK > 0) {
311  autoOKLabel = new uiLabel(panel, "", Point(4, y + 2));
312 
313  onTimer = [&](uiTimerHandle t) {
314  int now = esp_timer_get_time() / 1000;
315  if (lastUserActionTime() + 900 > now) {
316  killTimer(t);
317  destroyWindow(autoOKLabel);
318  return;
319  }
320  if (autoOK <= 0) {
321  killTimer(t);
322  retval = InputResult::Enter;
323  finalize();
324  }
325  --autoOK;
326  autoOKLabel->setTextFmt("%d", autoOK);
327  };
328  setTimer(this, 1000);
329 
330  }
331 
332  } else {
333  panel = nullptr;
334  }
335 
336  addControls();
337 
338  showWindow(mainFrame, true);
339  setActiveWindow(mainFrame);
340  if (controlToFocus)
341  setFocusedWindow(controlToFocus);
342 }

419 void SelectApp::calcRequiredSize()
420 {
421  auto messageExtent = canvas()->textExtent(font, messageText);
422  requiredWidth = imax(requiredWidth, messageExtent + 20);
423 
424  // calc space for message
425  requiredHeight += font->height;
426 
427  // calc space for list box
428  size_t maxLength;
429  auto itemsCount = countItems(&maxLength);
430  listBoxHeight = (font->height + 4) * itemsCount;
431  int requiredHeightUnCut = requiredHeight + listBoxHeight;
432  requiredHeight = imin(requiredHeightUnCut, canvas()->getHeight());
433  requiredWidth = imax(requiredWidth, maxLength * canvas()->textExtent(font, "M"));
434  if (requiredHeightUnCut > requiredHeight)
435  listBoxHeight -= requiredHeightUnCut - requiredHeight;
436 }

479 int SelectApp::countItems(size_t * maxLength)
480 {
481  *maxLength = 0;
482  int count = 0;
483  if (items) {
484  char const * start = items;
485  while (*start) {
486  auto end = strchr(start, separator);
487  if (!end)
488  end = strchr(start, 0);
489  int len = end - start;
490  *maxLength = imax(*maxLength, len);
491  start += len + (*end == 0 ? 0 : 1);
492  ++count;
493  }
494  } else if (itemsList) {
495  for (int i = 0; i < itemsList->count(); ++i)
496  *maxLength = imax(*maxLength, strlen(itemsList->get(i)));
497  count += itemsList->count();
498  }
499  return count;
500 }
