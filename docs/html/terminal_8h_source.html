843  void printf(const char * format, ...) {
844  va_list ap;
845  va_start(ap, format);
846  int size = vsnprintf(nullptr, 0, format, ap) + 1;
847  if (size > 0) {
848  va_end(ap);
849  va_start(ap, format);
850  char buf[size + 1];
851  auto l = vsnprintf(buf, size, format, ap);
852  write((uint8_t*)buf, l);
853  }
854  va_end(ap);
855  }
