411 auto wf = new T(waveform);
412 attach(wf);
413 wf->setFrequency(frequency);
414 wf->setVolume(volume);
415 wf->setAutoDestroy(true);
416 wf->setDuration(m_sampleRate / 1000 * durationMS);
